def calc_r_sqrt(x, y):
    x_sqrt = 0
    y_sqrt = 0
    xy = 0
    x_total = 0
    y_total = 0
    for i in range(0, len(x)):
        x_sqrt += x[i] ** 2
        y_sqrt += y[i] ** 2
        xy += x[i] * y[i]
        x_total += x[i]
        y_total += y[i]
    return ((len(x) * xy) - (x_sqrt * y_sqrt)) / (math.sqrt((len(x) * x_sqrt) - (x_total ** 2)) / math.sqrt((len(y) * y_sqrt) - (y_total ** 2))) ** 2

train = pd.read_csv('titanic/train.csv')
train.drop('Cabin', axis=1, inplace=True)
def add_age(cols):
    Age = cols[0]
    Pclass = cols[1]
    if pd.isnull(Age):
        return int(train[train["Pclass"] == Pclass]["Age"].mean())
    else:
        return Age
train["Age"] = train[["Age", "Pclass"]].apply(add_age,axis=1)
